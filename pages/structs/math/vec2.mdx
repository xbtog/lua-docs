### constructor

```lua
local v = vec2.new(2, 5)
```

### `+{:lua}`

```lua
local v = vec2.new(1, 2) + vec2.new(3, 2)

print(v) -- vec2(4, 4)
```

### `-{:lua}`

```lua
local v = vec2.new(1, 2) - vec2.new(1, 3)

print(v) -- vec2(0, -1)
```

### `*{:lua}`

```lua
local v = vec2.new(1, 2) * vec2.new(1, 3)

print(v) -- vec2(1, 6)
```

### `/{:lua}`

```lua
local v = vec2.new(2, 2) / vec2.new(2, 2)
local v2 = vec2.new(2, 2) / 2

print(v) -- vec2(1, 1)
print(v2) -- vec2(1, 1)
```
